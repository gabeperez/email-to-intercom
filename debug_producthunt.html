🔍 Product Hunt Email Detection Debug Tool

📋 Instructions:

  1. Go to the Product Hunt profile page with the email issue
  2. Open Developer Tools (F12)
  3. Go to Console tab
  4. Copy and paste the debug script below
  5. Run it to see what's happening with email detection

🐛 Debug Script:

// Product Hunt Email Detection Debug Script
console.log('🔍 Starting Product Hunt Email Detection Debug...');

// Check current URL
console.log('📍 Current URL:', window.location.href);

// Check if extension is loaded
const isExtensionLoaded = typeof chrome !== 'undefined' && chrome.runtime;
console.log('🔌 Extension loaded:', isExtensionLoaded);

// Define email regex (same as in extension)
const emailRegex = /\b[A-Za-z0-9]([A-Za-z0-9._%+-]*[A-Za-z0-9])?@[A-Za-z0-9]([A-Za-z0-9.-]*[A-Za-z0-9])?\.[A-Za-z]{2,7}\b/g;

// Find all potential email elements
const potentialElements = [];
const selectors = [
    'td span',
    'table td', 
    'tbody td',
    'tr td',
    'span',
    'td',
    'div'
];

selectors.forEach(selector => {
    const elements = document.querySelectorAll(selector);
    elements.forEach(element => {
        const text = element.textContent.trim();
        if (text.includes('@') && emailRegex.test(text)) {
            potentialElements.push({
                element: element,
                text: text,
                tagName: element.tagName,
                className: element.className,
                id: element.id
            });
        }
    });
});

console.log(`📧 Found ${potentialElements.length} potential email elements:`);
potentialElements.forEach((item, index) => {
    console.log(`${index + 1}.`, {
        text: item.text,
        tagName: item.tagName,
        className: item.className,
        id: item.id,
        element: item.element
    });
});

// Check for specific Product Hunt patterns
const verifiedEmailPattern = /([A-Za-z0-9._%+-]+@[A-Za-z0-9.-]+\.[A-Za-z|a-z]{2,7})\s*\(verified\)/g;
const pageText = document.body.textContent;
const verifiedMatches = [...pageText.matchAll(verifiedEmailPattern)];

console.log(`✅ Found ${verifiedMatches.length} verified email patterns:`);
verifiedMatches.forEach((match, index) => {
    console.log(`${index + 1}. Full match: "${match[0]}"`);
    console.log(`   Email: "${match[1]}"`);
});

// Check if any elements are already processed by extension
const processedElements = document.querySelectorAll('.email-intercom-processed');
console.log(`🔧 Found ${processedElements.length} elements already processed by extension:`);
processedElements.forEach((element, index) => {
    console.log(`${index + 1}.`, {
        text: element.textContent.trim(),
        tagName: element.tagName,
        element: element
    });
});

// Test the extension's email detection manually
console.log('🧪 Testing manual email detection...');
const testElement = potentialElements[0];
if (testElement) {
    console.log('Testing with element:', testElement.element);
    
    // Simulate the extension's validation
    const isValid = testElement.element.textContent.length <= 200 && 
                   testElement.element.children.length <= 5 &&
                   !testElement.element.classList.contains('email-intercom-processed');
    
    console.log('Element validation:', {
        textLength: testElement.element.textContent.length,
        childrenCount: testElement.element.children.length,
        alreadyProcessed: testElement.element.classList.contains('email-intercom-processed'),
        isValid: isValid
    });
}

console.log('✅ Debug complete! Check the results above.');
